import sys

def func():
    tokens = iter(sys.stdin.buffer.read().split())

    N = int(next(tokens))

    rel = []
    ids = []
    for _ in range(N):
        rel.append(float(next(tokens)))
        ids.append(int(next(tokens)))

    sim = []
    for _ in range(N):
        row = []
        for _ in range(N):
            row.append(float(next(tokens)))
        sim.append(row)

    lam = float(next(tokens))
    K = int(next(tokens))

    if K == 0:
        print()
        return

    selected = [False] * N
    max_sim = [0.0] * N   # 每个文档与已选集合中最相似文档的相似度
    answer = []
    eps = 1e-12

    for _ in range(K):
        best_idx = -1
        best_score = float("-inf")

        for i in range(N):
            if selected[i]:
                continue

            score = lam * rel[i] - (1 - lam) * max_sim[i]

            if score > best_score + eps:
                best_score = score
                best_idx = i
            elif abs(score - best_score) <= eps:
                if ids[i] < ids[best_idx]:
                    best_idx = i

        selected[best_idx] = True
        answer.append(str(ids[best_idx]))

        # 更新其余未选文档到已选集合的最大相似度
        for i in range(N):
            if not selected[i]:
                max_sim[i] = max(max_sim[i], sim[i][best_idx])

    print(" ".join(answer))

if __name__ == "__main__":
    func()
